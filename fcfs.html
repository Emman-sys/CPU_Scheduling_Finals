FCFS Algorithm

First-Come, First-Served (non-preemptive) - processes are scheduled in order of arrival.

/* FCFS pseudocode */
function calculateFCFS(processes) {
  // Sort by arrival time
  let sorted = processes.slice().sort((a,b) => a.at - b.at);
  let currentTime = 0;
  let results = [];

  for (let p of sorted) {
    if (currentTime < p.at) currentTime = p.at; // idle until arrival
    p.start = currentTime;
    currentTime += p.bt;
    p.ct = currentTime;
    p.tat = p.ct - p.at; // turnaround time
    p.wt = p.tat - p.bt; // waiting time
    results.push(p);
  }

  return results;
}
